#=========================== regression =======================================
REGRESSION
  CELLS_PER_PROCESS 4
END

#=========================== solver options ===================================
NEWTON_SOLVER FLOW
  ITOL_UPDATE 1.d0
END

LINEAR_SOLVER FLOW
  SOLVER DIRECT
END

#=========================== discretization ===================================
GRID
  TYPE STRUCTURED
  ORIGIN 0.d0 0.d0 0.d0
  NXYZ 1 1 10
  BOUNDS
    0.d0 0.d0 0.d0
    1.d0 1.d0 10.d0
  /
END

#=========================== characteristic curves ============================
CHARACTERISTIC_CURVES default
  SATURATION_FUNCTION VAN_GENUCHTEN
    M 0.5d0
    ALPHA  1.d-4
    LIQUID_RESIDUAL_SATURATION 0.1d0
    MAX_CAPILLARY_PRESSURE 1.d8
  /
  PERMEABILITY_FUNCTION MUALEM_VG_LIQ
    M 0.5d0
    LIQUID_RESIDUAL_SATURATION 0.1d0
  /
END

#=========================== output options ===================================
OUTPUT
  VARIABLES
    LIQUID_PRESSURE
    POROSITY
    PERMEABILITY
  /
END

#=========================== times ============================================
TIME
  FINAL_TIME 1. y
  INITIAL_TIMESTEP_SIZE 0.1d0 y
  MAXIMUM_TIMESTEP_SIZE 1.d0 y
END

#=========================== regions ==========================================
REGION all
  COORDINATES
    -1.d20 -1.d20 -1.d20
    1.d20 1.d20 1.d20
  /
END

REGION bottom
  FACE BOTTOM
  COORDINATES
    -1.d20 -1.d20 0.d0
    1.d20 1.d20 0.d0
  /
END

REGION top
  FACE TOP
  COORDINATES
    -1.d20 -1.d20 10.d0
    1.d20 1.d20 10.d0
  /
END

#=========================== flow conditions ==================================
FLOW_CONDITION initial
  TYPE
    PRESSURE HYDROSTATIC
  /
  DATUM 0.d0 0.d0 1.d0
  PRESSURE 101325.d0
END

FLOW_CONDITION infiltration
  TYPE
    FLUX neumann
  /
  FLUX 10 cm/yr
END

#=========================== condition couplers ===============================
# initial condition
INITIAL_CONDITION
  FLOW_CONDITION initial
  REGION all
END

# bottom boundary condition
BOUNDARY_CONDITION bottom
  FLOW_CONDITION initial
  REGION bottom
END

BOUNDARY_CONDITION infiltrometer
  FLOW_CONDITION infiltration
  REGION top
END

#=========================== stratigraphy couplers ============================
STRATA
  REGION all
  MATERIAL soil1
END
